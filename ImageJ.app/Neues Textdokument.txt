from pydicom import dcmread
path = 'INPUT/IM-0001-0001.dcm'
ds =dcmread(path)
print(ds)
sn= "\n"

moda = str(ds.Modality)
manu = str(ds.Manufacturer)
instname = str(ds.InstitutionName)
instaddress =str(ds.InstitutionAddress)
physname= str(ds.ReferringPhysicianName)
studdisc=str(ds.StudyDescription)
seriesdisc=str(ds.SeriesDescription)
pn = str(ds.PatientName)
pID= str(ds.PatientID)
pbd=str(ds[0x10,0x30].value)
th=str(ds[0x18,0x1130].value)
ps=str(ds.PatientSex)
output = moda+sn+manu+sn+instname+sn+instaddress+sn+physname+sn+studdisc+sn+seriesdisc+sn+pn+sn+pID+sn+pbd+sn+ps+sn+th
print(output, file = open('OUTPUT/TEST.txt', 'w'))
